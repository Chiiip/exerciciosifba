public class JavaApplication2 {

/*
Escreva uma Programa que receba 10 números inteiros e imprima a quantidade de números primos dentre os números que foram digitados.  

*/
 
    public static void main(String[] args) {
    String num;
int vnum, cont = 1, primo = 0, somaprimo = 0,cont2 = 1;
while (cont2<=10){
num = JOptionPane.showInputDialog("Insira um número para verificar se é primo ou não");
vnum = Integer.parseInt(num);
cont = 1; 
primo = 0;
while (cont<=vnum){
if (vnum%cont == 0){
    primo++;
}    
cont++;
}
if (primo <=2){
  somaprimo = somaprimo + 1;
}
cont2++;
}
  JOptionPane.showMessageDialog(null, "O número de primos é " + somaprimo);  
    }
}
